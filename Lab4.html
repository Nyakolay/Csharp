        int n;
        do
        {
            Console.Write("Введите кол-во элементов списка(>5) -> ");
            n = int.Parse(Console.ReadLine());
        } while (n < 5);
        List mas1 = new List();
        for (int i = 0; i < n; i++)
        {
            mas1.Add((char)random.Next(0x0410, 0x44F));
        }
        List mas2 = new List();
        for (int i = 0; i < (mas1.Count / 2); i++)
        {
            mas2.Add(mas1[i]);
        }
        mas2.Sort();
        foreach (char i in mas2)
        {
            Console.WriteLine(i);
        }
        int count = -1;
        try
        {
            Console.Write("\nВывести элемент под номером -> ");
            count = int.Parse(Console.ReadLine());

        {

        {
