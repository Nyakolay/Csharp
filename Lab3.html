abstract class K1
{
    public abstract void vvod();
    public abstract void vivod();
}
class K2 : K1
{
    private double x, y, z;
    public K2()
    {
        this.x = 0;
        this.y = 0;
        this.z = 0;
    }

    public override void vvod()
    {
        Console.Write("Введите x: ");
        x = double.Parse(Console.ReadLine());
        Console.Write("Введите y: ");
        y = double.Parse(Console.ReadLine());
        Console.Write("Введите z: ");
        z = double.Parse(Console.ReadLine());
    }
    public override void vivod()
    {
        Console.WriteLine("Вывод данных объекта класса К2: x = {0}, y = {1}, z = {2}", x, y, z);
    }
}
class K3 : K1
{
    private int a, b, c;
    public K3() { a = 0; b = 0; c = 0; }
    public override void vvod()

        Console.Write("Введите a: ");

        Console.Write("Введите b: ");
        b = int.Parse(Console.ReadLine());
        Console.Write("Введите c: ");
        c = int.Parse(Console.ReadLine());
    }
    public override void vivod()
    {
        Console.WriteLine("Вывод данных объекта класса К3: a = {0}, b = {1}, c = {2}", a, b, c);
    }
}

class Program
{
    static void Main(string[] args)
    {
        K1 xyz = new K2();

interface IMain
{
    void VvodData(IGetData data);
    void VivodData(IGetData data);
}

interface IGetData
{
    void vivod();
    void vvod();
}

class k : IMain
{
    public void VivodData(IGetData data)
    {
        data.vivod();
    }

    public void VvodData(IGetData data)
    {
        data.vvod();
    }
}

interface IK1
{
    void vvod();
    void vivod();
}
interface IK2 : IK1
{
    void sum();
    void mult();
}
class myclass : IK2
{
    private int x, y, z;
    public void mult()
    {
        Console.WriteLine("Произведение x * y * z = {0} * {1} * {2} = {3}\n", x, y, z, (x * y * z));
    }

    public void sum()
    {
        Console.WriteLine("Сумма x + y + z = {0} + {1} + {2} = {3}\n", x, y, z, (x + y + z));
    }

    public void vivod()
    {
        Console.WriteLine("Вывод данных: x = {0}, y = {1}, z = {2}\n", x, y, z);
    }

    public void vvod()
    {
        Console.WriteLine("Введите данные:");
        Console.Write("x -> ");
        x = int.Parse(Console.ReadLine());
        Console.Write("y -> ");
        y = int.Parse(Console.ReadLine());
        Console.Write("z -> ");
        z = int.Parse(Console.ReadLine());
        Console.WriteLine();
    }
}

internal class Program
{
    private static void Main(string[] args)
    {
        myclass myclass = new myclass();
        myclass.vvod();
        myclass.vivod();
        myclass.sum();
        myclass.mult();
    }
}
